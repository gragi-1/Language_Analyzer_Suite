// Sistema m√≠nimo con funciones
function void mostrar(int num) {
    write 'Valor: ';
    write num;
}

function int incrementar(int x) {
    let int nuevo;
    if (x < 10) nuevo = x + 1;
    return nuevo;
}

let int cont = 5;
mostrar(cont);

if (cont < 8) {
    cont = incrementar(cont);
    mostrar(cont);
} else {
    write 'Alto';
}

cont += 2;
write 'Final: ';
write cont;
