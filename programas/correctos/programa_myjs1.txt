// Función simple de comparación
function int comparar(int x, int y) {
    let int resultado;
    if (x < y)
        resultado = x + y;
    return resultado;
}

let int a = 5;
let int b = 3;
let int r = comparar(a, b);

if (r < 10) {
    write 'Resultado pequeño';
} else {
    write 'Resultado grande';
}

a += 2;
write a;
