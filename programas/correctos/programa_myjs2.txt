// Programa 2: Sistema de calificaciones
let int nota = 75;
let string mensaje;

if (nota >= 70) {
    if (nota >= 90) {
        mensaje = 'Excelente';
    } else {
        mensaje = 'Aprobado';
        nota += 5; // Bono por esfuerzo
    }
} else {
    mensaje = 'Reprobado';
}
write 'Estado: ';
write mensaje;
write 'Nota final: ';
write nota;
