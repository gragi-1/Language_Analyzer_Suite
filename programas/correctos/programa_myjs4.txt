// Función validación simple
function boolean esValido(int valor) {
    let boolean ok;
    if (valor < 0) {
        ok = false;
    } else {
        ok = true;
    }
    return ok;
}

let int dato;
read dato;
let boolean val = esValido(dato);

if (val)
    write 'Aceptado';

if (dato < 100 && val) {
    dato += 10;
    write 'Ajustado';
}

write dato;
