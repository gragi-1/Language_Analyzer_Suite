// Funci√≥n suma con if-else
function int sumar(int x, int y) {
    let int total;
    if (x < y) {
        total = x + y;
    } else {
        total = y + x;
    }
    return total;
}

let int n1 = 8;
let int n2 = 4;
let int res = sumar(n1, n2);

if (res < 20) {
    write 'Res bajo: ';
    write res;
}

n2 += 3;
write 'n2: ';
write n2;
