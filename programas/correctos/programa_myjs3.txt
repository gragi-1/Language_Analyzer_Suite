// Programa 3: Manipulación de texto
let string saludo = 'Hola';
let string nombre = 'Juan';
let string mensajeCompleto;

if (nombre != '') {
    mensajeCompleto = saludo;
    mensajeCompleto += ', '; // Concatenamos
    mensajeCompleto += nombre;
    mensajeCompleto += '!';
} else {
    mensajeCompleto = 'Hola anónimo!';
}
write mensajeCompleto;
// Fin del programa
